
Lecture 05　元素(1): 圖像、列表、表格

--------------------------------------------------------------------------------

<!DOCTYPE html>
<html>
<head>
    <meta content="text/html; charset=utf-8">
    <title>Sample13</title>
</head>

<body>
    <img src="xxx">　<!--內聯層級不自動換行，需定義br或下個段落-->

    <p><img src="xxx" width="180" height="120" align="top">apple</p>　<!--align預設bottom，垂直對齊會換行-->
    <p><img src="xxx" width="180" height="120" style="float: right">apple</p>　<!--設定漂移位置(文字仍為靠左): 等同align="right"水平對齊-->

    <p><a href="yyy"><img src="xxx"></p>　<!--在圖片設定超連結yyy網址; img, a均無結束標籤-->
</body>
</html>

<!DOCTYPE html>
<html>
<head>
    <meta content="text/html; charset=utf-8">
    <title>Sample14</title>
</head>

<body>
    <h3>這是兩個無序列表</h3>
    <ul><li>我是第一項</li><li>我是第二項</li></ul>
    <ul style="list-style-type: square"><li>我是第三項</li><li>我是第四項</li></ul>　<!--預設為disc，circle為非實心圓-->

    <h3>這是兩個有序列表</h3>
    <ol><li>我是第五項</li><li>我是第六項</li></ol>　<!--標籤改用ol-->
    <ol type=I start=7><li>我是第七項</li><li>我是第八項</li></ol>　<!--type: A/a, I/i，以start="數字"指定起始項標序-->

    <h3>這是一個巢狀列表</h3>
    <ul><li>我是第九項
    <ul><li>我是第一小項</li><li>我是第二小項</li></ul></li>　<!--巢狀包覆第一項內，需加寫ul標籤；預設為非實心圓-->
    <li>我是第十項</li></ul>
</body>
</html>

<!DOCTYPE html>
<html>
<head>
    <meta content="text/html; charset=utf-8">
    <title>Sample15</title>
</head>

<body>
    <table border=1>　<!--預設border=0(無表格邊框)-->
    <thead>　<!--可統一設定表頭、表身格式，於此例可略-->
        <tr><th>x<0</th><th>x>0</th></tr>　<!--表頭欄位為th標籤，自動設定粗體置中-->
    </thead>
    <tbody>
        <tr><td>我是第二象限 (-,+)</td><td>我是第一象限 (+,+)</td></tr>
        <tr><td>我是第三象限 (-,-)</td><td>我是第四象限 (+,-)</td></tr>
    </tbody>
    </table>
</body>
</html>

<!DOCTYPE html>
<html>
<head>
    <meta content="text/html; charset=utf-8">
    <title>Sample16</title>
</head>

<body>
    <table border=1 cellpadding=15 cellspacing=5>　<!--單元格邊距對內(文字); 單元格間距對外(表格)-->
    <h3>垂直表頭寫法</h3>　<!--無法對齊，可用caption(預設置中)style="margin-bottom"取代-->
        <tr><th>y>0</th><td>我是第二象限 (-,+)</td><td>我是第一象限 (+,+)</td></tr>
        <tr><th>y<0</th><td>我是第三象限 (-,-)</td><td>我是第四象限 (+,-)</td></tr>
    </table>
</body>
</html>

    <table border=1 cellpadding=8 style="text-align: center">　<!--定義: 所有元素置中對齊-->
        <tr><th/><th colspan=2>x軸</th></tr>
        <tr><th rowspan=2>y軸</th><td> II </td><td> I </td></tr>
        <tr><td> III </td><td> IV </td></tr>　<!--首欄為rowspan範圍，直接輸入另兩欄即可-->
    </table>
</body>
</html>
