
Lecture 03　美化: CSS選擇器、顏色

--------------------------------------------------------------------------------

<!DOCTYPE html>
<html>
<head>
    <meta content="text/html; charset=utf-8">
    <title>Sample06</title>
</head>

<body style="background-color: blue">　<!--套用於整個區塊-->
    <p style="font-size: 40px; color: yellow">我是段落1</p>　<!--包覆於style="值"之內，內部則以冒號、分號為格式-->
    <h2 style="background-color: #FF0000">我是標題2</h2>　<!--font-family定義字形，仍可定義區塊背景色-->
</body>
</html>


--------------------------------------------------------------------------------

<!DOCTYPE html>
<html>
<head>
    <meta content="text/html; charset=utf-8">
    <title>Sample07</title>
    <style>
    p{font-family: 標楷體; color: purple}
    h2{background-color: pink; text-align: center}　/* 於style定義: 內部樣式表* /
    </style>
</head>

<body style="background-color: #0000FF">　<!--直接定義: 內聯樣式，亦可以rgb(0, 0, 255)表之-->
    <p>我是段落1</p>
    <h2>我是標題2</h2>
</body>
</html>


--------------------------------------------------------------------------------

<!DOCTYPE html>
<html>
<head>
    <meta content="text/html; charset=utf-8">
    <title>Sample08</title>
    <style>
    body{background-color: rgb(192, 192, 192)}
    .style{font-size: 15px; text-align: center}　/* 類別選擇器: 不具唯一性 */
    p.style{color: green}　/* 類別選擇器可加上標籤額外指定 */
    #another{background-color: green}　/* ID選擇器: 具有唯一性 */
    </style>
</head>

<body>
    <p class="style">我是段落1</p>　<!--受到body, .style 和p.style影響-->
    <h1 class="style">我是標題1</h1>　<!--受到body, .style影響-->
    <h2 id="another">我是標題2</h2>　<!--定義條件衝突時，以後者為準-->
</body>
</html>
