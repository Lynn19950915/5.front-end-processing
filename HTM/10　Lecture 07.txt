<!DOCTYPE html>
<html>
<head>
    <meta content="text/html; charset=utf-8">
    <title>Sample22</title>
</head>

<body>
    <p><i>這是一行<del>粗</del>斜體字</i></p>　<!--p相對div具行距特色-->
    <p><b>這是一行粗體字，可設定<sub>下標</sub>與<sup>上標</sup></b></p>
    <p><u>這是一行底線字，可以<big>放大</big>也能<small>縮小</small></u></p>

    <p><bdo dir="rtl">反過來</bdo></p>　<!--顛倒輸出以bdo定義，rtl表由右至左-->
</body>
</html>

<!DOCTYPE html>
<html>
<head>
    <meta content="text/html; charset=utf-8">
    <title>Sample25</title>
</head>

<body>
    <p id="click">可寫入一段事件腳本</p>

    <script>　<!--事件觸發是作用在 function，可全寫在同一腳本-->
    function event1( )
    {document.getElementById("click").style.color="#ff0000"}　<!--針對id=click元素調整樣式-->
    function event2( )
    {document.getElementById("click").innerHTML=("Magic")}　<!--針對id=click元素輸出HTML(內容)-->
    </script>

    <button type="button" onclick="event1( )">幫我修改顏色</button>　<!--不同於input按鈕，button功能視事件自定義-->　
    <button type="button" onclick="event2( )">幫我換一句話</button>
</body>
</html>
